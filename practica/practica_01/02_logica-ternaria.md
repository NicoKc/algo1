## Ejercicio 16

- a) (¬x ∨L b)

	- (¬x ∨L b)
	- (¬⊥ ∨L true)
	- ⊥

- b) ((c ∨L (y ∧L a)) ∨ b)

	- ((c ∨L (y ∧L a)) ∨ b)
	- ((true ∨L (⊥ ∧L false)) ∨ true)
	- (true ∨ true)

- c) ¬(c ∨L y)

	- ¬(c ∨L y)
	- ¬(true ∨L ⊥)
	- ¬(true)

	- false

- d) (¬(c ∨L y) ↔ (¬c ∧L ¬y))

	- (¬(c ∨L y) ↔ (¬c ∧L ¬y))
	- (¬(true ∨L ⊥) ↔ (¬true ∧L ¬⊥))
	- (¬true ↔ false)
	- false ↔ false
	- true

- e) ((c ∨L y) ∧L (a ∨L b))

	- ((c ∨L y) ∧L (a ∨L b))
	- ((true ∨L ⊥) ∧L (false ∨L true))
	- (true ∧L true)
	- true

- f) (((c ∨L y) ∧L (a ∨L b)) ↔L (c ∨L (y ∧L a) ∨L b))

	- (((c ∨L y) ∧L (a ∨L b)) ↔L (c ∨L (y ∧L a) ∨L b))
	- (((true ∨L ⊥) ∧L (false ∨L true)) ↔L (true ∨L (⊥ ∧L false) ∨L true))
	- ((true ∧L true) ↔L (true ∨L (⊥ ∧L false) ∨L true))
	- ( true ↔L (true ∨L (⊥ ∧L false) ∨L true))
	- ( true ↔L ((true ∨L ⊥) ∨L true))
	- ( true ↔L true)
	- true

- g) (¬c ∧L ¬y)

	- (¬c ∧L ¬y)
	- (¬true ∧L ¬⊥)
	- (false ∧L ¬⊥)
	- (false ∧L ⊥)
	- ⊥
