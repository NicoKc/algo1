    \item $Q_{c}    \equiv \longitud{r} = \longitud{R_{0}} \wedge (\paraTodo{j}{0}{\longitud{a}} \implicaLuego r[j] = a[j]) \wedge (\paraTodo{j}{0}{\longitud{b}} \implicaLuego r[j+\longitud{a}] = b[j])$
