proc ralentizar(inout r: reunion, in prof: Z, in freq: Z) {
    Pre { esReuniónVálidaAux(r, prof, freq) ∧ r0 = r }
    Post {
        esReuniónVálidaAux(r, prof, freq) ∧L (
            lasSeñalesTienenElDobleDeMuestras(r, r0) ∧
            promedioEntrePares(r, r0)
        )
    }
}

pred lasSeñalesTienenElDobleDeMuestras(in r: reunion, in r0: reunion) {
    (∀i : Z) (
        (0 ≤ i < |r0|) →L (
            2*|fst r0[0]| = |fst r[i]| + 1
        )
    )
}

pred promedioEntrePares(in r: reunion, in r0: reunion) {
    (∀i : Z) (
        (0 ≤ i < |r|) →L (
            (∃j : Z) (
                (0 ≤ j < |r0|) →L (
                    (snd r[i] = snd r0[i]) ∧L (
                            (∀q : Z) (
                                ((0 < q < |fst r[j]|)) (
                                if (esPar(q)) then
                                    (fst r[j])[q] = (fst r0[i])[q/2]
                                else 
                                    (fst r[j])[q] = ((fst r0[i])[(q-1)/2]+(fst r0[i])[(q+1)/2])/2
                            )

                    )
                )
            )
        )
    )
}
