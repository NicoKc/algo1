\begin{proc}{esSeñal}{\In s: \TLista{\ent}, \In prof: \ent, \In freq: \ent, out result: \bool}{}
\pre{  prof > 0 \wedge freq > 0 }
\post{ result = \True \Iff esSeñalAux(s, prof, freq) }
\end{proc}

\pred{esSeñalAux}{s: \TLista{\ent}, prof: \ent, freq: \ent}{

    $\quad\longitud{s} \geq 0 \wedge$

    $\quad\quad frecuenciaEnRango (freq) \wedge$

    $\quad\quad profundidadCorrecta (s) \wedge$

    $\quad\quad duraMasDeUnSegundo (s, freq) \wedge$

    $\quad\quad ningunaMuestraSuperaLaProfundidad(s, prof)$

}

$ $

\pred{frecuenciaEnRango}{$freq$: \ent}{ freq \in [8,32] }

$ $

\pred{profundidadCorrecta}{$prof$: \ent}{ prof \in [8,16,32] }

$ $

\pred{duraMasDeUnSegundo}{$s$: \TLista{\ent}, $freq$: \ent}{
    duraciónEnSegundos(s, freq) > 1
}

$ $

\pred{ningunaMuestraSuperaLaProfundidad}{s: \TLista{\ent}, p: \ent}{

    $\quad\paraTodo{i}{0}{|s|} \implicaLuego (-2)^{p-1} \leq s[i] \leq 2^{p-1}-1$

}

$ $

\aux{duraciónEnSegundos}{$s$: \TLista{\ent}, $freq$: \ent}{\ent}{\frac{\longitud{s}}{(freq \cdot 1000)}}
