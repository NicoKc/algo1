proc ordenar(inout r: reunion, in freq: Z, in prof : Z) {
    Pre { esReuniónVálida(r, prof, freq) }
    Post {
        (∀i : Z) (
            (0 ≤ i < |r|) →L (
                laSeñalEstaOrdenada(snd r[i])
            )
        )
    }
}

pred laSeñalEstaOrdenada(s: señal) {
    (∀i: Z)(
        (0 ≤ i < |r|) →L (
            s[i] >= s[i + 1]
        )
    )
}
